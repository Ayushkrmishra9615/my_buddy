  function displayMaterials() {
    let filtered = allMaterials.filter((m, i) => {
      const matchFilter = activeFilter === 'all' || m.type === activeFilter;
      const matchSearch = !searchQuery ||

        m.sem.toLowerCase().includes(searchQuery);
